void playerM(char *spaces, std::string name, char player);
void computerM(char *spaces, char comp);
bool winner(char *spaces, char player);
bool tie(char *spaces);

int main() {
    
    std::string Pname;
    char spaces[9] = {' ', ' ', ' ', ' ', ' ', ' ', ' ', ' ', ' '};
    char player = 'X';
    char comp = 'O';
    bool run = true;
    
    srand(time(0));
    
    details(Pname);
    grid(spaces);
    
    while(run){
        
        playerM(spaces, Pname, player);
        grid(spaces);
        if(winner(spaces, player)){
            run = false;
            break;
        }
        else if(tie(spaces)){
            run = false;
            break;
        }
        
        computerM(spaces, comp);
        grid(spaces);
        if(winner(spaces, player)){
            run = false;
            break;
        }
        else if(tie(spaces)){
            run = false;
            break;
        }
    }
    return 0;
}

void playerM(char *spaces, const std::string name, char player){
    
    int number;
    
    do{
        std::cout << name << " enter numbers from (1-9): ";
        std::cin >> number;
        number--;
        if(spaces[number] == ' '){
            spaces[number] = player;
            break;
        }
    }while(!number > 0 || !number < 8);
}
void computerM(char *spaces, char comp){
    
    int number;
    
    while(true){
        number = rand() % 9;
        int n = number+1;

        if(spaces[number] == ' '){
            std::cout << "Computer entered: " << n;
            spaces[number] = comp;
            break;
        }
    }
}

bool tie(char *spaces){
    
    for(int i=0; i<9; i++){
        if(spaces[i]==' '){
            return false;
        }
    }
    std::cout << "IT'S A TIE!\n";
    return true;
}
